CREATE DATABASE cowlife;

-------

CREATE TABLE scenes
(
id INT(25) UNIQUE PRIMARY KEY,
name VARCHAR(30) NOT NULL UNIQUE,
picturelink TEXT(240) NOT NULL,
objectid SMALLINT(25),
object2id SMALLINT(25),
object3id SMALLINT(25),
object4id SMALLINT(25),
object5id SMALLINT(25),
position1x SMALLINT(9000),
position2x SMALLINT(9000),
position3x SMALLINT(9000),
position4x SMALLINT(9000),
position5x SMALLINT(9000),
position1y SMALLINT(9000),
position2y SMALLINT(9000),
position3y SMALLINT(9000),
position4y SMALLINT(9000),
position5y SMALLINT(9000)
);

-----

CREATE TABLE objects
(
id SMALLINT(25) UNIQUE NOT NULL PRIMARY KEY,
picturelink TEXT(240)NOT NULL,
eventid SMALLINT(25),
name VARCHAR(30)NOT NULL UNIQUE,
);


-----

CREATE TABLE events
(
id SMALLINT(25)NOT NULL UNIQUE PRIMARY KEY,
name VARCHAR(30)NOT NULL UNIQUE,
code TEXT(500),
);
