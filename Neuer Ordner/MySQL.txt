CREATE DATABASE cowlife;

-------

CREATE TABLE scenes
(
id SMALLINT,
name VARCHAR,
picturelink TEXT,
objectid SMALLINT,
object2id SMALLINT,
object3id SMALLINT,
object4id SMALLINT,
object5id SMALLINT,
position1x SMALLINT,
position2x SMALLINT,
position3x SMALLINT,
position4x SMALLINT,
position5x SMALLINT,
position1y SMALLINT,
position2y SMALLINT,
position3y SMALLINT,
position4y SMALLINT,
position5y SMALLINT,
UNIQUE(id, name),
PRIMARY KEY (id)
);

-----

CREATE TABLE objects
(
id SMALLINT(size),
picturelink TEXT,
eventid SMALLINT(size),
name VARCHAR(size),
position SMALLINT,
UNIQUE(id, name),
PRIMARY KEY (id)
);


-----

CREATE TABLE events
(
ID SMALLINT(size),
NAME VARCHAR(size),
CODE MEDIUMTEXT,
UNIQUE(id, name),
PRIMARY KEY (id)
);
